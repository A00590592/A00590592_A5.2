def build_price_catalog(price_data):
    """Builds a price catalog mapping title -> price.
       Returns a dictionary and a checkpoint message."""
    catalog = {}

    if not isinstance(price_data, list):
        print("Error: Price catalogue JSON must be a list.")
        return catalog

    for idx, item in enumerate(price_data, start=1):
        if not isinstance(item, dict):
            print(f"Invalid product entry at index {idx}")
            continue

        title = item.get("title")
        price = item.get("price")

        if isinstance(title, str) and isinstance(price, (int, float)):
            catalog[title] = float(price)
        else:
            print(f"Invalid product data at index {idx}")

    print(f"CHECKPOINT: Catalog entries loaded: {len(catalog)}")
    return catalog
